update_party_popularity = {
	set_local_variable = {
		name = popularity_accumulator
		value = 0
	}
	every_member = {
		leader = {
			change_local_variable = {
				name = popularity_accumulator
				add = popularity
			}
		}
	}
	root.set_variable = {
		name = var:$NAME$_popularity
	}
}

update_party_clout = {
	set_local_variable = {
		name = ig_clout_accumulator
		value = 0
	}
	every_member = {
		change_local_variable = {
			name = ig_clout_accumulator
			add = ig_clout
		}
	}
	root.set_variable = {
		name = var:$NAME$_clout
	}
}

update_winner_morale = {
	root.change_variable = {
		name = var:$WINNER$_morale
		add = 25
	}
}

update_loser_morale = {
	root.change_variable = {
		name = var:$LOSER$_morale
		add = -25
	}
}
